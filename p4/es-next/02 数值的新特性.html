# 数值的新特性

# 数值运算

// 1 二八禁止表示法
Number('0b111')  // 7 ， 二进制转换
Number('0o10')  // 8， 八进制转换

// 2 全局上有方法可以直接使用
parseInt('12.34') // 12
parseFloat('123.45#') // 123.45

// ES6 更加模块化
Number.parseInt('12.34') // 12
Number.parseFloat('123.45#') // 123.45

// 3 小数点位数
(123.45).tofixed(0) = '123'

// 4 截取整数部分
Math.trunc(4.1) // 4
Math.trunc(-4.9) // -4

// 5 指数运算
2 ** 2 // 4

// 6 幂运算
const V2 = Math.pow(3, 3) // 3 * 3 * 3

// 7 对数运算
Math.log(1024) / Math.log(2) // 1024 是 2 的 10 次方

# 数值判断

// 1 判断整数
Number.isInteger(25) // true
Number.isInteger(25.1) // false

// 2 正负数判断
Math.sign(-5) // -1
Math.sign(5) // +1
Math.sign(0) // +0
Math.sign(-0) // -0
Math.sign(NaN) // NaN

// 3 浮点数计算 Number.EPSILO 是一个很小的数
function withinErrorMargin (left, right) {
  return Math.abs(left - right) < Number.EPSILON * Math.pow(2, 2);
}

0.1 + 0.2 === 0.3 // false， 浮点数直接比较，因有误差所以不相等
withinErrorMargin(0.1 + 0.2, 0.3) // true

1.1 + 1.3 === 2.4 // false
withinErrorMargin(1.1 + 1.3, 2.4) // true
